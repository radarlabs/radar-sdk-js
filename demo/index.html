Initialize a new map with the Radar SDK and your publishable key, and render it on the page.

              <!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8" />
    <meta name="viewport" content="initial-scale=1,maximum-scale=1,user-scalable=no" />
    <link href="https://js.radar.com/v4.0.0/radar.css" rel="stylesheet">
    <script src="https://js.radar.com/v4.0.0/radar.js"></script>
  </head>

  <body>
    <div id="map-container">
      <div id="map" style="width: 100%; height: 500px;" />
    </div>

    <script type="text/javascript">
      Radar.initialize('prj_test_pk_...');

      Radar.ui.map({
        container: 'map',
      });
    </script>
  </body>
</html>
